/*1. Programar un algoritmo recursivo que permita hacer la divisi√≥n  por restas sucesivas:*/

public class GarciaEduard_ex1{
    /**
     * @author GarciaEduard
     * @param x
     * @param j
     * @return
     */
    public static int divisions_rest(int x, int j) {
        if (x < j) {
            return 0;
        }
        return 1 + divisions_rest(x - j, j);
    }



    /**
     * @author GarciaEduard
     * @param args
     */
    public static void main(String[] args) {
        int resp = divisions_rest(10, 3);
        System.out.println(resp);
    }

}
